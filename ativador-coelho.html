🛡️ Regras de Segurança — Firestore

Copie estas regras para o seu console Firebase → Firestore → Regras:

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // Usuários: só o próprio usuário acessa seus dados
    match /users/{userId} {
      allow read, write: if request.auth != null
                         && request.auth.uid == userId;

      match /workouts/{workoutId} {
        allow read, write: if request.auth != null
                           && request.auth.uid == userId;
      }
      match /exercises/{exerciseId} {
        allow read, write: if request.auth != null
                           && request.auth.uid == userId;
      }
      match /history/{historyId} {
        allow read, write: if request.auth != null
                           && request.auth.uid == userId;
      }
    }

    // Tokens: qualquer usuário autenticado pode LER (verificar)
    // mas NUNCA criar ou editar via cliente (só pelo admin SDK)
    match /tokens/{tokenId} {
      allow read: if request.auth != null;
      // Permite que o cliente marque o token como usado (update)
      // mas só se o token ainda não foi usado
      allow update: if request.auth != null
                    && resource.data.used == false;
      // Nenhum cliente pode criar ou deletar tokens
      allow create, delete: if false;
    }
  }
}
